// Helper for number validation (no additional libraries!)
bool isNumber(const string& s) {
    if (s.empty()) return false;
    for (size_t i = 0; i < s.size(); ++i)
        if (!isdigit(s[i])) return false;
    return true;
}

void payUtilityBill(const string& type, int minLen, int maxLen) {
    string id;
    double amount;
    if(selectedLanguage == 2) {
        cout << "\nDarj karein " << type << " Reference/Customer Number (" << minLen << "-" << maxLen << " hase)";
    } else {
        cout << "\nEnter " << type << " Reference/Customer Number (" << minLen << "-" << maxLen << " digits): ";
    }
    cin >> id;
    int len = id.length();
    if (len < minLen || len > maxLen) {
        setColor(12);
        if(selectedLanguage == 2) {
            cout << "\nGhalat " << type << " Reference/Customer Number!";
        } else {
            cout << "\nInvalid " << type << " Reference/Customer Number!";
        }

    }
    if(selectedLanguage == 2) {
        cout << "Bill ki raqam darj karein: $";
    } else {
        cout << "Enter Bill Amount: $";
    }
    cin >> amount;

int enterPIN() {
    clearScreen();
    showHeader();

    string inputPin;
    char ch;
    if(selectedLanguage == 2) {
        cout << "Apna 4 hase ka PIN darj karein: ";
    } else {
        cout << "Enter your 4-digit PIN: ";
    }
    while (inputPin.length() < 4) {
        ch = _getch();
        if (ch >= '0' && ch <= '9') {
            inputPin.push_back(ch);
            cout << '*';
        } else if (ch == '\b' && !inputPin.empty()) {
            inputPin.pop_back();
            cout << "\b \b";
        }
    }
    cout << endl;

    if (inputPin == pass) {
        setColor(10);
        if(selectedLanguage == 2) {
            cout << "\nDakhla ki izin mil gayi.";
        } else {
            cout << "\nAccess granted.";
        }
        Sleep(1500);
        return 1;
    } else {
        setColor(12);
        if(selectedLanguage == 2) {
            cout << "\nPIN ghalat hai.";
        } else {
            cout << "\nIncorrect PIN.";
        }
        Sleep(1500);
        return 0;
    }
}
